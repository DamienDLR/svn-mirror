\documentclass[twocolumn]{article}
\usepackage{expl3}
\begin{document}
This test deals with messages in the info file and 
console output; nothing to see here in the text.
\CodeStart

\tlp_new:N  \err_lbl
\tlp_new:N  \err_lbl_a
\tlp_new:N  \err_lbl_b

\exp_args:NNx \err_file_new:Nn \c_errfile_tlp {\c_job_name_tlp.err}

\tlp_set:Nn \err_lbl_a {ERR-ONE}
\err_interrupt_new:NNNnnn \c_errfile_tlp \err_lbl_a 1
 {This~is~an~error~with~one~arg:~`#1'}
 {\err_help_return_or_X:}
 {}

\tlp_set:Nn \err_lbl_b {ERR-TWO}
\err_interrupt_new:NNNnnn \c_errfile_tlp \err_lbl_b 2
 {This~is~a~different~error~with~two~args:~`#1'~;~`#2'}
 {\err_help_return_or_X:}
 {}

\err_file_close:N \c_errfile_tlp

% A more friendly wrapper for "\err_interrupt:NNw"
\def_new:NNn \err_interrupt:nnw 2 {
  \tlp_set:Nn \err_lbl {#2}
  \exp_args:Nc \err_interrupt:NNw {c_#1_tlp} \err_lbl
}

%% Should pause and print:
%%   This is an error with one arg: ` hello '
\err_interrupt:nnw {errfile} {ERR-ONE} { hello }

%% Should pause and print:
%%   This is an error with two args: ` hello? ' ; ` oh, hello '
\err_interrupt:nnw {errfile} {ERR-TWO} { hello? } { oh, hello. }

\CodeStop
\end{document}
