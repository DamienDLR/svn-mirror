\documentclass[twocolumn]{article}
\usepackage{expl3}
\begin{document}
This test deals with messages in the info file and 
console output; nothing to see here in the text.
\ExplSyntaxOn

\exp_args:NNx \err_file_new:Nn \c_errfile_tl {\c_job_name_tl.err}

% First error message
\tl_new:N  \err_lbl_a
\tl_set:Nn \err_lbl_a {err1arg}
\err_interrupt_new:NNpnnn \c_errfile_tl \err_lbl_a #1
 {This~is~an~error~with~one~arg:~`#1'}
 {\err_help_return_or_X:}
 {}

% Second error message
\tl_new:N  \err_lbl_b
\tl_set:Nn \err_lbl_b {err2arg}
\err_interrupt_new:NNpnnn \c_errfile_tl \err_lbl_b #2
 {This~is~a~different~error~with~two~args:~`#1'~;~`#2'}
 {\err_help_return_or_X:}
 {}

\err_file_close:N \c_errfile_tl

% A more friendly wrapper for "\err_interrupt:NNw"
\tl_new:N  \err_lbl
\cs_set:Npn \err_interrupt:nnw #1#2 {
  \tl_set:Nn \err_lbl {#2}
  \exp_args:Nc \err_interrupt:NNw {c_#1_tl} \err_lbl
}

%% Should pause and print:
%%   This is an error with one arg: ` hello '
\err_interrupt:nnw {errfile} {err1arg} { hello }

%% Should pause and print:
%%   This is an error with two args: ` hello? ' ; ` oh, hello '
\err_interrupt:nnw {errfile} {err2arg} { hello? } { oh, hello. }

% (In a package/class you'd use a more descriptive name than "errfile".)

\ExplSyntaxOff
\end{document}
